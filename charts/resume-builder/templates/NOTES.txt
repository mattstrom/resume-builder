MongoDB Backup:
  Backups run on schedule: {{ .Values.mongodbBackup.schedule }}
  PVC: {{ include "resume-builder.fullname" . }}-mongodb-backup

  To restore from the latest backup:
    helm upgrade {{ .Release.Name }} ./charts/resume-builder \
      --set mongodbRestore.enabled=true \
      --set mongodbRestore.backupFile=latest

  To restore from a specific backup:
    helm upgrade {{ .Release.Name }} ./charts/resume-builder \
      --set mongodbRestore.enabled=true \
      --set mongodbRestore.backupFile=mongodb_backup_YYYYMMDD_HHMMSS.tar.gz

  After restore completes, disable the restore Job:
    helm upgrade {{ .Release.Name }} ./charts/resume-builder \
      --set mongodbRestore.enabled=false
{{- end }}
